00779 void aravisCamera::run() {
00780     epicsTimeStamp lastFeatureGet, now;
00781     int getFeatures;
00782     ArvBuffer *buffer;
00783 
00784     /* Wait for database to be up */
00785     while (!iocRunning) {
00786         epicsThreadSleep(0.1);
00787     }
00788 
00789     /* Loop forever */
00790     epicsTimeGetCurrent(&lastFeatureGet);
00791     while (1) {
00792         /* Wait 5ms for an array to arrive from the queue */
00793         if (epicsMessageQueueReceiveWithTimeout(this->msgQId, &buffer, sizeof(&buffer), 0.005) == -1) {
00794             /* No array, so if there is a camera, get the next feature*/
00795             if (this->camera != NULL && this->connectionValid == 1) {
00796                 /* We only want to get a feature once every 25ms (max 40 features/s)
00797                  * so compare timestamp against last feature get
00798                  */
00799                 epicsTimeGetCurrent(&now);
00800                 if (epicsTimeDiffInSeconds(&now, &lastFeatureGet) > 0.025) {
00801                     this->lock();
00802                     getIntegerParam(AravisGetFeatures, &getFeatures);
00803                     if (getFeatures) {
00804                         this->getNextFeature();
00805                         callParamCallbacks();
00806                     }
00807                     this->unlock();
00808                 }
00809             }
00810         } else {
00811             /* Got a buffer, so lock up and process it */
00812             this->lock();
00813             this->processBuffer(buffer);
00814 
00815             /* free memory */
00816             g_object_unref(buffer);
00817             if (this->stream != NULL) {
00818                 /* Allocate the new raw buffer we use to compute images. */
00819                 this->allocBuffer();
00820             }
00821             this->unlock();
00822         }
00823     }

00853     getIntegerParam(ADNumImagesCounter, &numImagesCounter);

00957                 }

00993     }
00994
